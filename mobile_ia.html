if (forfaitsFiltres.length === 0) { console.log("⚠️ Aucun forfait trouvé, élargissement des critères..."); forfaitsFiltres = forfaits.filter(function(forfait) { return forfait.prixPromo <= budgetMax * 1.2; }); }

// **Toujours aucun forfait ? Retour des moins chers**

if (forfaitsFiltres.length === 0) { console.log("🚨 Aucun forfait trouvé, affichage des moins chers."); return forfaits.sort(function(a, b) { return a.prixPromo - b.prixPromo }).slice(0, 5); }

// **Scoring standard**

forfaitsFiltres.forEach(function(forfait) { forfait.score = 0; var ecartData = Math.abs(forfait.data - besoinDataTotal); forfait.score += Math.max(50 - ecartData, 0); if (reponses.appels === "2") forfait.score += 15; if (reponses.sms === "2") forfait.score += 10; if (reponses.voyage === "3" && forfait.data_europe_DOM) { var dataEurope = parseInt(forfait.data_europe_DOM) || 0; forfait.score += dataEurope >= 50 ? 30 : 15; } forfait.score += Math.max(30 - Math.abs(budgetMax - forfait.prixPromo) * 2, 0); });

// **Trier et retourner les 5 meilleurs forfaits**

var resultats = forfaitsFiltres.sort(function(a, b) { return b.score - a.score }).slice(0, 5); console.log("✅ Forfaits sélectionnés :", resultats); return resultats;

 }
